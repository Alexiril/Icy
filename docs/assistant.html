106    def handle_request(self, state: "AppState", request: str) -> None:
107        if request == "":
108            return
109        words: list[str] = request.lower().split(" ")
110        if not self.chat_mode and self.name.lower() not in words:
111            return
112        self.current_request = request
113        print(colored(f'{translations["Request:"]} {request}', "blue"))
114        task = "chat"
115        if len(words) == 1:
116            task = "activate"
117            if (
118                function := self.capabilities.get(
119                    "activate", AssistantFunction([], None)
120                ).reaction
121            ) is not None:
122                function(state)
123        elif len(words) >= 2:
124            for command_end in range(1, len(words) + 1):
125                if (
126                    task := state.intention_classifier.classify(
127                        request=(" ".join(words[:command_end])).strip()
128                    )
129                ) is not None:
130                    if (function := self.capabilities[task].reaction) is not None:
131                        function(state, *words)
132                    break
133            else:
134                if self.use_chat:
135                    task = "chat"
136                    self.task_chat(state, *words)
137                elif (function := self.failure) is not None:
138                    function(state)
139        print(colored(f'{translations["Classified task:"]} {task}', "yellow"))

165    def task_bye(self, state: "AppState", *args: Any) -> None:
166        state.voice_handler.say(
167            choice(
168                [
169                    translations["See you next time."],
170                    translations["Okay, see you."],
171                    translations["Okay, bye!"],
172                    translations["Have a good day."],
173                ]
174            ),
175        )
176        if len(list(self.ai_messages.queue)) > 0:
177            general_data = ""
178            for chunk in self.handle_gpt(
179                state,
180                "Select the most import data from our conversation and "
181                "print it as short and detailed as possible.",
182                list(self.ai_messages.queue),
183                translations[
184                    "You are a simple text parser. Don't answer as an assistant. "
185                    "Answer exactly how you have to."
186                ],
187            ):
188                if isinstance(chunk, ChatCompletionChunk):
189                    chunk = chunk.choices[0].delta.content
190                    if chunk is None:
191                        break
192                general_data += chunk
193            config: dict[str, Any] = {}
194            if exists("prev.data"):
195                with open("prev.data", "rt") as file:
196                    config.update(loads(file.read()))
197            config["gpt_info"] = general_data
198            with open("prev.data", "wt") as file:
199                print(JSONEncoder().encode(config), file=file)
200        raise VoiceHandler.StopRecording()

227    def handle_gpt(
228        self,
229        state: "AppState",
230        request: str,
231        history: list[ChatCompletionMessageParam],
232        system_prompt: str | None = None,
233        add_gpt_info: bool = True,
234    ) -> Iterable[str] | Stream[ChatCompletionChunk]:
235        history = history.copy()
236        bot_name_description = translations[
237            "You are a nice friend, not an assistant, your name is"
238        ]
239        final_command = translations["Answer creative and detailed, but short."]
240        history.insert(
241            0,
242            (
243                {
244                    "role": "system",
245                    "content": (
246                        (
247                            (
248                                f'{bot_name_description} '
249                                f'{self.name}. {translations["You are a"]} '
250                                f'{translations[self.gender]}. '
251                                f'{final_command} '
252                            )
253                            if system_prompt is None
254                            else system_prompt
255                        )
256                        + f'{translations["You know that:"]
257                             if len(self.gpt_info) != 0
258                             else ""} {self.gpt_info}'
259                        if add_gpt_info
260                        else ""
261                    ),
262                }
263            ),
264        )
265
266        def offline_gpt() -> Iterable[str]:
267            def gpt_brakes(token_id: int, response: str) -> bool:
268                return state.voice_handler.speak_worker.work
269
270            with self.gpt_model.chat_session():
271                self.gpt_model._history = history  # type: ignore
272                return self.gpt_model.generate(
273                    request,
274                    max_tokens=1024,
275                    temp=1.0,
276                    streaming=True,
277                    callback=gpt_brakes,
278                )
279
280        if not state.settings.use_openai_gpt:
281            return offline_gpt()
282        try:
283            return OpenAI().chat.completions.create(
284                model="gpt-4o-mini",
285                messages=history + [{"role": "user", "content": request}],
286                temperature=1,
287                stream=True,
288            )
289        except APIStatusError:
290            print(colored("OpenAI services are unavaliable.", "light_red"))
291        except Exception:
292            print_traceback()
293        return offline_gpt()

108        if request == "":

111        if not self.chat_mode and self.name.lower() not in words:
112            return
113        self.current_request = request
114        print(colored(f'{translations["Request:"]} {request}', "blue"))
115        task = "chat"

119                function := self.capabilities.get(

125            for command_end in range(1, len(words) + 1):
126                if (
127                    task := state.intention_classifier.classify(
128                        request=(" ".join(words[:command_end])).strip()
129                    )
130                ) is not None:
131                    if (function := self.capabilities[task].reaction) is not None:
132                        function(state, *words)
133                    break
134            else:
135                if self.use_chat:
136                    task = "chat"
137                    self.task_chat(state, *words)
138                elif (function := self.failure) is not None:
139                    function(state)
140        print(colored(f'{translations["Classified task:"]} {task}', "yellow"))

229        self,

235    ) -> Iterable[str] | Stream[ChatCompletionChunk]:
236        history = history.copy()
